27template <NumericOrComplex T, StorageOrder S>
28template <bool B>
29COO<T, S>::COO(std::bool_constant<B>, SizetPairContainer auto const& indexes,
30 NumericContainer auto const& values) {
31#ifdef DEBUG
32 size_t index_size =
33 static_cast<size_t>(std::distance(indexes.begin(), indexes.end()));
34 size_t values_size =
35 static_cast<size_t>(std::distance(values.begin(), values.end()));
36 assert(index_size == values_size &&
37 "Error in COO constructor: sizes don't match.\n");
38#endif
39
40 std::vector<T> tempvalues(values.begin(), values.end());
41 std::vector<std::pair<size_t, size_t>> tempindexes(indexes.begin(),
42 indexes.end());
43
44 comparator = Comparator<S>{};
45 std::vector<size_t> p = sort_permutation(tempindexes, comparator);
46
47 apply_permutation(tempindexes, p);
48 apply_permutation(tempvalues, p);
49
50 indexptr = std::make_unique<indexlist>();
51 valuesptr = std::make_unique<valueslist>();
52
53 size_t max_index_r = 0, max_index_c = 0;
54 auto it1 = tempindexes.rbegin();
55 auto it2 = tempvalues.rbegin();
56
57 while (it1 != tempindexes.rend() && it2 != tempvalues.rend()) {
58#ifdef DEBUG
59 if constexpr (B) {
60 assert(
61 (*it1).first < this->rows && (*it1).second < this->columns &&
62 "Error in COO constructor: indexes out of bounds (too big).\n");
63 }
64 assert((*it1).first >= 0 && (*it1).second >= 0 &&
65 "Error in COO constructor: indexes out of bounds (negative).\n");
66
67 if (it1 != tempindexes.rbegin()) {
68 auto compare = it1;
69 --compare;
70 assert((*it1) != (*compare) &&
71 "Error in COO constructor: redefinition of the same element "
72 "(equal indexes).\n");
73 }
74#endif
75
76 if constexpr (!B) {
77 max_index_r = std::max(max_index_r, (*it1).first);
78 max_index_c = std::max(max_index_c, (*it1).second);
79 }
80
81 indexptr->push_front(*it1);
82 valuesptr->push_front(*it2);
83 it1++;
84 it2++;
85 }
86
87 if constexpr (!B) this->resize(max_index_r + 1, max_index_c + 1);
88}

243template <NumericOrComplex T, StorageOrder S>
244T COO<T, S>::find_dynamic_const(size_t i, size_t j) const {
245 auto it1 = indexptr->begin();
246 auto it2 = valuesptr->begin();
247
248 while (it1 != indexptr->end()) {
249 if ((*it1).first == i && (*it1).second == j) {
250 return *it2;
251 }
252 else if (!comparator(*it1, std::make_pair(i, j)))
253 break;
254
255 it1++;
256 it2++;
257 }
258
259 return 0;

270template <NumericOrComplex T, StorageOrder S>
271T& COO<T, S>::find_dynamic(size_t i, size_t j) {
272 auto it1 = indexptr->begin();
273 auto next1 = indexptr->before_begin();
274
275 auto it2 = valuesptr->begin();
276 auto next2 = valuesptr->before_begin();
277
278 while (it1 != indexptr->end()) {
279 if ((*it1).first == i && (*it1).second == j) {
280 return *it2;
281 }
282 else if (!comparator(*it1, std::make_pair(i, j))) {
283 indexptr->insert_after(next1, std::make_pair(i, j));
284 it2 = valuesptr->insert_after(next2, T{});
285 return *it2;
286 }
287
288 it1++;
289 next1++;
290 it2++;
291 next2++;
292 }
293
294 indexptr->insert_after(next1, std::make_pair(i, j));
295 it2 = valuesptr->insert_after(next2, T{});
296 return *it2;
297}

405template <NumericOrComplex T, StorageOrder S>
406template <NormType N>
407double COO<T, S>::norm_dynamic() const {
408 if constexpr ((N == Infinity && S == rowMajor) ||
409 (N == One && S == columnMajor)) {
410 double res = 0.0;
411 double sum = 0.0;
412 int i = 0;
413
414 auto it2 = (*valuesptr).begin();
415 for (auto it = (*indexptr).begin(); it != (*indexptr).end(); ++it) {
416 if (comparator((*it), next_line(i))) {
417 sum += std::abs(*it2);
418 }
419 else {
420 res = std::max(sum, res);
421 i++;
422 sum = std::abs(*it2);
423 }
424 it2++;
425 }
426 res = std::max(sum, res);
427 return res;
428 }
429
430 else if constexpr (N == One && S == rowMajor) {
431 std::vector<double> partial_res(this->columns, 0);
432
433 auto it2 = (*valuesptr).begin();
434 for (auto it = (*indexptr).begin(); it != (*indexptr).end(); ++it) {
435 partial_res[(*it).second] += std::abs(*it2);
436 it2++;
437 }
438 return *std::max_element(partial_res.begin(), partial_res.end());
439 }
440 else if constexpr (N == Infinity && S == columnMajor) {
441 std::vector<double> partial_res(this->rows, 0);
442
443 auto it2 = (*valuesptr).begin();
444 for (auto it = (*indexptr).begin(); it != (*indexptr).end(); ++it) {
445 partial_res[(*it).first] += std::abs(*it2);
446 it2++;
447 }
448 return *std::max_element(partial_res.begin(), partial_res.end());
449 }
450 else {
451 double sum = 0.0;
452
453 for (auto it = (*valuesptr).begin(); it != (*valuesptr).end(); ++it) {
454 sum += std::abs(*it) * std::abs(*it);
455 }
456 return std::sqrt(sum);
457 }
458}

468template <NumericOrComplex T, StorageOrder S>
469bool COO<T, S>::remove_dynamic(size_t i, size_t j) {
470 auto prev1 = (*indexptr).before_begin();
471 auto prev2 = (*valuesptr).before_begin();
472
473 for (auto it1 = (*indexptr).begin(); it1 != (*indexptr).end(); ++it1) {
474 if ((*it1).first == i && (*it1).second == j) {
475 (*indexptr).erase_after(prev1);
476 (*valuesptr).erase_after(prev2);
477 return true;
478 }
479 ++prev1;
480 ++prev2;
481 }
482 return false;
483}

544template <NumericOrComplex T, StorageOrder S>
545std::vector<T> by_vector_dynamic(COO<T, S> const& m, std::vector<T> const& v) {
546 std::vector<T> result(m.rows);
547
548 auto indexit = m.indexptr->begin();
549 auto valuesit = m.valuesptr->begin();
550
551 while (indexit != m.indexptr->end()) {
552 result[indexit->first] += *valuesit * v[indexit->second];
553 ++indexit;
554 ++valuesit;
555 }
556
557 return std::move(result);
558}
