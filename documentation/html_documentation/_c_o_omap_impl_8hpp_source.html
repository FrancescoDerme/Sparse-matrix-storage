25template <NumericOrComplex T, StorageOrder S>
26template <bool B>
27COOmap<T, S>::COOmap(std::bool_constant<B>,
28 SizetPairContainer auto const& indexes,
29 NumericContainer auto const& values) {
30#ifdef DEBUG
31 size_t index_size =
32 static_cast<size_t>(std::distance(indexes.begin(), indexes.end()));
33 size_t values_size =
34 static_cast<size_t>(std::distance(values.begin(), values.end()));
35 assert(index_size == values_size &&
36 "Error in COOmap constructor: sizes don't match.\n");
37#endif
38
39 size_t max_index_r = 0, max_index_c = 0;
40 matrixptr = std::make_unique<valuesmap>();
41
42 auto it2 = values.begin();
43 for (auto it1 = indexes.begin(); it1 != indexes.end(); ++it1) {
44#ifdef DEBUG
45 if constexpr (B) {
46 assert((*it1).first < this->rows && (*it1).second < this->columns &&
47 "Error in COOmap constructor: indexes out of bounds (too "
48 "big).\n");
49 }
50
51 assert(
52 (*it1).first >= 0 && (*it1).second >= 0 &&
53 "Error in COOmap constructor: indexes out of bounds (negative).\n");
54
55 assert((*matrixptr).find(*it1) == (*matrixptr).end() &&
56 "Error in COOmap constructor: redefinition of the same element "
57 "(equal indexes).\n");
58#endif
59
60 if constexpr (!B) {
61 max_index_r = std::max(max_index_r, (*it1).first);
62 max_index_c = std::max(max_index_c, (*it1).second);
63 }
64
65 (*matrixptr)[*it1] = *it2;
66 it2++;
67 }
68
69 if constexpr (!B) this->resize(max_index_r + 1, max_index_c + 1);
70}

119template <NumericOrComplex T, StorageOrder S>
120COOmap<T, S>::COOmap(std::string& file_name) {
121 std::ifstream file(file_name);
122#ifdef DEBUG
123 assert(file && "Cannot open file.");
124#endif
125
126 std::string line;
127 std::getline(file, line);
128#ifdef DEBUG
129 assert(line.substr(0, 14) == "%%MatrixMarket" &&
130 "Invalid Matrix Market file.");
131#endif
132
133 while (std::getline(file, line)) {
134 if (line.empty()) continue;
135 if (line[0] != '%') break;
136 }
137
138 std::istringstream dims(line);
139 int nnz;
140 dims >> this->rows >> this->columns >> nnz;
141
142 valuesmap m;
143 size_t row, col;
144 T value;
145 for (int i = 0; i < nnz && std::getline(file, line); ++i) {
146 std::istringstream dims(line);
147 dims >> row >> col >> value;
148 m[{row - 1, col - 1}] = value;
149 }
150
151 this->matrixptr = std::make_unique<valuesmap>(std::move(m));
152}

257template <NumericOrComplex T, StorageOrder S>
258template <NormType N>
259double COOmap<T, S>::norm_dynamic() const {
260 if constexpr (N == Infinity) {
261 std::vector<double> partial_res(this->rows, 0);
262
263 for (auto& el : *matrixptr) {
264 partial_res[el.first.first] += std::abs(el.second);
265 }
266 return *std::max_element(partial_res.begin(), partial_res.end());
267 }
268
269 else if constexpr (N == One) {
270 std::vector<double> partial_res(this->columns, 0);
271
272 for (auto& el : *matrixptr) {
273 partial_res[el.first.second] += std::abs(el.second);
274 }
275 return *std::max_element(partial_res.begin(), partial_res.end());
276 }
277 else {
278 double sum = 0.0;
279
280 for (auto& el : *matrixptr) {
281 sum += std::abs(el.second) * std::abs(el.second);
282 }
283 return std::sqrt(sum);
284 }
285}

352template <NumericOrComplex T, StorageOrder S>
353std::vector<T> by_vector_dynamic(COOmap<T, S> const& m,
354 std::vector<T> const& v) {
355 std::vector<T> result(m.rows);
356 auto mapit = m.matrixptr->begin();
357
358 while (mapit != m.matrixptr->end()) {
359 result[mapit->first.first] += mapit->second * v[mapit->first.second];
360 ++mapit;
361 }
362
363 return std::move(result);
